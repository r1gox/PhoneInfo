# Installation
    pkg update && pkg upgrade -y
    pkg install -y curl
    pkg install -y git
    git clone https://github.com/Rigo-Jimenez/PhoneInfo
    cd PhoneInfo
    ls
    chmod 777 PhoneInfo.sh
    bash PhoneInfo.sh [Disfruta]
    bash PhoneInfo.sh --info [Informacion del creador]
